# waymark

Simple, lightweight, strongly typed router for React

```ts
const routes = [
  "/a/:b/c",
  "/:a/:b/*",
  "/a/b/:c",
  "/:a/b/c",
  "/a/:b/:c",
  "/:a/:b/c",
  "/a/b/c",
  "/a/*",
  "/a/b/*",
  "/:a/b/*",
  "/*",
  "/a/:b/*",
  "/:a/:b/:c",
  "/:a/*",
  "/:a/b/:c"
];
```
